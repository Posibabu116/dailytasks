1. Program to find the sum of all even numbers in an array using a for loop 


let arr = [1, 2, 3, 4, 5, 6];
let sum = 0;

for (let i = 0; i < arr.length; i++) {
    if (arr[i] % 2 === 0) {
        sum += arr[i];
    }
}

console.log("Sum of all even numbers: " + sum);

Output:


Sum of all even numbers: 12

2. Using a do-while loop to print numbers from 1 to 10, but not equal to 5 


function printNumbers() {
    let num = 1;
    do {
        if (num !== 5) {
            console.log(num);
        }
        num++;
    } while (num <= 10);
}

3. Check whether the given input string is a palindrome or not


function isPalindrome(str) {
    let reversedStr = str.split('').reverse().join('');
    if (str === reversedStr) {
        console.log(str + " is a palindrome");
    } else {
        console.log(str + " is not a palindrome");
    }
}

isPalindrome("mom");
isPalindrome("father");

Output:


mom is a palindrome
father is not a palindrome

5. Using a while loop to find the average of all numbers in an array 


let arr = [1, 2, 3, 4, 5, 6];
let sum = 0;
let i = 0;

while (i < arr.length) {
    sum += arr[i];
    i++;
}

let average = sum / arr.length;
console.log("Average: " + average);

if (average > 50) {
    console.log("Average is large");
}

Output:


Average: 3.5
