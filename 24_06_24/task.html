Array Methods Tasks

Task 1: Using concat()

// Merge two arrays into one.
const array1 = [1, 2, 3];
const array2 = [4, 5, 6];
let mergedArray = array1.concat(array2);
console.log(mergedArray);
Output: 
[1, 2, 3, 4, 5, 6]

Task 2: Using flat()

// Flatten a nested array up to a specific depth.
const nestedArray = [1, [2, [3, [4]]]];
let flattenedArray = nestedArray.flat(2);
console.log(flattenedArray);
Output: 
[1, 2, 3, [4]]

Task 3: Using splice()

// Remove elements from an array and optionally insert new elements.
const array = [1, 2, 3, 4, 5];
let removedElements = array.splice(2, 2, 'a', 'b');
console.log(array);
console.log(removedElements);
Output: 
[1, 2, 'a', 'b', 5]
[3, 4]

Task 4: Using slice()

// Extract a portion of an array into a new array.
const array = [1, 2, 3, 4, 5];
let slicedArray = array.slice(1, 4);
console.log(slicedArray);
Output: 
[2, 3, 4]

Task 5: Using indexOf()

Task 6: Using copyWithin()

// Copy part of an array to another location in the same array.
const array = [1, 2, 3, 4, 5];
array.copyWithin(0, 3, 5);
console.log(array);
Output: 
[4, 5, 3, 4, 5]

Task 10: Combining Methods

// Use multiple array methods in a single task.
const array1 = [1, 2, 3];
const array2 = [4, 5, 6];
let mergedArray = array1.concat(array2).flat();
console.log(mergedArray);

let splicedArray = mergedArray.slice(1, 5).toSpliced(2, 1, 'a');
console.log(splicedArray);
console.log(splicedArray.includes('a')); // true
console.log(splicedArray.indexOf(5)); // 3
Output: 
[2, 3, 4, 5, 'a', 6]
true
3
