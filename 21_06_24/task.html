JavaScript Functions and Outputs

Function demo(a, b)

// Returns the sum of two numbers.
function demo(a, b) {
    return a + b;
}
console.log(demo(10, 10));
Output: 
20

Function demo1(b, c)

// Returns the difference between two numbers.
function demo1(b, c) {
    return b - c;
}
console.log(demo1(5, 3));
Output: 
2

Function demo2(d, e)

// Returns the product of two numbers.

Function square(sqa, sqb)

// Returns the power of a number.
function square(sqa, sqb) {
    return sqa ** sqb;
}
console.log(square(3, 4));
Output: 
81

Function square1(num)

// Returns the square of a number.
function square1(num) {
    let sqvalue = num ** 2;
    return sqvalue;
}
console.log(square1(2));
Output: 
4

Function highest(x, y)

// Returns the higher value between two numbers.
function highest(x, y) {
    return x > y ? x : y;
}
console.log(highest(10, 20));
Output: 
20

Function interest(p, r, t)

// Calculates total value including interest.
function interest(p, r, t) {
    let interest = (p * r * t) / 100;
    let total = p + interest;
    return total;
}
console.log(interest(10000, 10, 3));
Output: 
13000
