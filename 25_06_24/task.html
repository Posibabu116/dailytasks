Array of Objects Functions

Function 1: getKeysArray(arrayOfObjects)

// Returns an array of keys of each object in the input array.
function getKeysArray(arrayOfObjects) {
    let keysArray = [];
    for (let obj of arrayOfObjects) {
        let keys = [];
        for (let key in obj) {
            keys.push(key);
        }
        keysArray.push(keys);
    }
    return keysArray;
}

let input1 = [{ a: 1, b: 2 }, { c: 3, d: 4 }, { e: 5, f: 6 }];
console.log(getKeysArray(input1));
Output: 
[["a", "b"], ["c", "d"], ["e", "f"]]

Function 2: getValuesArray(arrayOfObjects)

// Returns an array of values of each object in the input array.
function getValuesArray(arrayOfObjects) {
    let valuesArray = [];
    for (let obj of arrayOfObjects) {
        let values = [];
        for (let key in obj) {
            values.push(obj[key]);
        }
        valuesArray.push(values);
    }
    return valuesArray;
}

let input2 = [{ a: 1, b: 2 }, { c: 3, d: 4 }, { e: 5, f: 6 }];
console.log(getValuesArray(input2));
Output: 
[[1, 2], [3, 4], [5, 6]]

Function 3: filterObjectsByProperty(arrayOfObjects, property)

// Returns an array of objects that have a specific property.
function filterObjectsByProperty(arrayOfObjects, property) {
    let filteredArray = [];
    for (let obj of arrayOfObjects) {
        if (obj.hasOwnProperty(property)) {
            filteredArray.push(obj);
        }
    }
    return filteredArray;
}

let input3 = [{ a: 1 }, { b: 2 }, { a: 3, b: 4 }];
console.log(filterObjectsByProperty(input3, 'a'));
Output: 
[{ a: 1 }, { a: 3, b: 4 }]

Function 4: sortObjectsByProperty(arrayOfObjects, property)

// Returns an array of objects sorted by a specific property.
function sortObjectsByProperty(arrayOfObjects, property) {
    arrayOfObjects.sort((a, b) => a[property] - b[property]);
    return arrayOfObjects;
}

let input4 = [{ a: 2 }, { a: 1 }, { a: 3 }];
console.log(sortObjectsByProperty(input4, 'a'));
Output: 
[{ a: 1 }, { a: 2 }, { a: 3 }]

Function 5: mergeObjectsProperties(arrayOfObjects)

// Returns an object with properties from all objects in the input array.
function mergeObjectsProperties(arrayOfObjects) {
    let mergedObject = {};
    for (let obj of arrayOfObjects) {
        for (let key in obj) {
            mergedObject[key] = obj[key];
        }
    }
    return mergedObject;
}

let input5 = [{ a: 1 }, { b: 2 }, { c: 3 }];
console.log(mergeObjectsProperties(input5));
Output: 
{ a: 1, b: 2, c: 3 }
