Print before letters of every character using ASCII values

let inputString = "hello";
let result = "";

for (let i = 0; i < inputString.length; i++) {
  let currentChar = inputString[i];
  let previousChar = String.fromCharCode(currentChar.charCodeAt(0) - 1);
  result += previousChar;
}

console.log(result);

Remove spaces between a sentence

let sentence = "p o s i";
let res = "";

for (let i = 0; i < sentence.length; i++) {
  if (sentence[i] !== ' ') {
    res += sentence[i];
  }
}

Reverse a word without using methods

let word = "hello";
let reversed = "";

for (let i = word.length - 1; i >= 0; i--) {
  reversed += word[i];
}

console.log(reversed);
