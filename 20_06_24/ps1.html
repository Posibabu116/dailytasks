Iterating Over Objects in JavaScript

Question 1: Iterate over an object using for-in loop to display keys

let person = { "firstName": "Alice", "lastName": "Johnson", "occupation": "Engineer" }; for (let key in person) { console.log(key); }

Question 2: Iterate over an object using for-in loop to display values

let person = { "firstName": "Alice", "lastName": "Johnson", "occupation": "Engineer" }; for (let key in person) { console.log(person[key]); }

Question 3: Iterate over an object using for-in loop to display both keys and values

let person = { "firstName": "Alice", "lastName": "Johnson", "occupation": "Engineer" }; for (let key in person) { console.log(key + " : " + person[key]); }

Question 4: Iterate over an object using for-of loop to display keys

let employee = { "empId": "E123", "empName": "Bob Smith", "department": "HR" }; for (let [key] of Object.entries(employee)) { console.log(key); }

Question 5: Iterate over an object using for-of loop to display both keys and values

let employee = { "empId": "E123", "empName": "Bob Smith", "department": "HR" }; for (let [key, value] of Object.entries(employee)) { console.log(key + " : " + value); }
